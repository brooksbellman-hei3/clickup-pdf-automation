version: '3.8'

services:
  clickup-reporter:
    build: 
      context: .
      dockerfile: Dockerfile
    container_name: clickup-pdf-reporter
    restart: unless-stopped
    environment:
      # ClickUp API
      - CLICKUP_API_TOKEN=${CLICKUP_API_TOKEN}
      - CLICKUP_LIST_ID=${CLICKUP_LIST_ID}
      
      # Email Configuration
      - SMTP_HOST=${SMTP_HOST}
      - SMTP_PORT=${SMTP_PORT}
      - SMTP_USER=${SMTP_USER}
      - SMTP_PASS=${SMTP_PASS}
      - EMAIL_TO=${EMAIL_TO}
      
      # Scheduling
      - TIMEZONE=${TIMEZONE:-America/New_York}
      - SEND_HOUR=${SEND_HOUR:-9}
    volumes:
      # Mount logs directory (optional)
      - ./logs:/usr/src/app/logs
    networks:
      - clickup-network
    healthcheck:
      test: ["CMD", "node", "-e", "console.log('Health check')"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 60s

networks:
  clickup-network:
    driver: bridge
